Please try again"); } else if(amt > curr_bal){ System.out.println("Insufficient Balance! Please try again"); } else{ curr_bal -= amt; count++; System.out.println("Successfully withdrawed P" + amt); System.out.println("New Balance: P" + curr_bal); } break; case 4: System.out.println("Exiting...."); is_running = false; break; } } System.out.println("Total transaction Count: " + count); System.out.println("Balance: P" + curr_bal); } }

import java.util.Scanner ; public class SimpleWalletCortezo { public static void main ( String [] args ) { // Setup scanner to read user input Scanner scanner = new Scanner ( System . in ) ; // Starting balance int balance = 3000 ; // This counts how many successful transactions happened int count = 0; // This controls the while loop // If true = keep looping, if false = stop boolean running = true ; // Keep showing the menu until the user picks Exit while ( running ) { // Show the menu System . out . println ( " \ n===WALLET MENU===" ) ; System . out . println ( "1. Check Balance" ) ; System . out . println ( "2. Add Money" ) ; System . out . println ( "3. Spend Money" ) ; System . out . println ( "4. Exit" ) ; System . out . print ( "Choice: " ) ; // Read what the user picked int choice = scanner . nextInt () ; // Check what the user picked using switch - case switch ( choice ) { case 1: // User wants to check balance System . out . println ( "Output: Balance = " + balance ) ; count = count + 1; // add 1 to the counter break ; case 2: // User wants to add money System . out . print ( "Add: " ) ; int addAmount = scanner . nextInt () ; // Check if the amount is valid if ( addAmount > 0) { // Valid amount, so add it to balance balance = balance + addAmount ; System . out . println ( "Output: New Balance = " + balance ) ; count = count + 1; // add 1 to the counter } else { // Invalid amount System . out . println ( " Output: Invalid amount. Must be greater than 0." ) ; } break ; case 3: // User wants to spend money System . out . print ( "Spend: " ) ; int spendAmount = scanner . nextInt () ; // Rule 1: Amount must be greater than 0 if ( spendAmount <= 0) { System . out . println ( " Output: Invalid amount. Must be greater than 0." ) ; // Rule 2: Cannot spend more than 1000 pesos at a time } else if ( spendAmount > 1000 ) { System . out . println ( " Output: Exceeds maximum spend" ) ; // Rule 3: Cannot spend more than current balance } else if ( spendAmount > balance ) { System . out . println ( " Output: Insufficient balance" ) ; } else { // All rules passed, spending is allowed balance = balance - spendAmount ; System . out . println ( " Output: New Balance = " + balance ) ; count = count + 1; // add 1 to the counter } break ; case 4: // User wants to exit System . out . println ( "Transaction Count: " + count ) ; System . out . println ( "Exiting program..." ) ; running = false ; // this stops the while loop break ; default : // User typed something that is not 1, 2, 3, or 4 System . out . println ( "Output: Invalid choice. Please select 1 - 4." ) ; } } // end of while loop // Close the scanner when done } }
Reflection
Working on the Simple Wallet System was the exact moment where the core concepts of Object-Oriented Programming finally clicked for me. Designing a distinct Wallet class equipped with specific methods like deposit and withdraw felt incredibly natural and practical. It was fascinating to see how we can model real-world objects and behaviors directly as Java classes. This activity boosted my confidence significantly because programming stopped feeling like abstract math and started feeling like building actual, usable tools.

import java.util.Scanner; public class ExpenseTrackerCortezo { // this is the void method that displays the program title public static void showTitle() { System.out.println(" Personal Expense Tracker "); } // this is the non-void method that calculates the total of all expenses public static double computeTotal(double food, double transportation, double other, double savings) { double total = food + transportation + other + savings; return total; } // this is the non-void method that checks if total exceeds the budget public static String checkBudget(double total, double budget) { if (total > budget) { return "Over Budget!"; } else { return "Within Budget!"; } } // this is the void method that displays the total expense and budget status public static void showResult(String name, double food, double transportation, double other, double savings, double total, double budget) { System.out.println("\n--- Expense Summary for " + name + " ---"); System.out.println("Food: Php" + food); System.out.println("Transportation: Php" + transportation); System.out.println("Other: Php" + other); System.out.println("Savings: Php" + savings); System.out.printf("Total Expenses: Php%.2f%n", total); System.out.printf("Budget: Php%.2f%n", budget); // compute remaining budget double remaining = budget - total; System.out.printf("Remaining Budget: Php%.2f%n", remaining); // budget status String status = checkBudget(total, budget); System.out.println("Budget Status: " + status); // personalized message if (total > budget) { System.out.println("\nIt seems you went all over your budget. Try to spend less next time, " + name + "!"); } else { System.out.println("\nWow! It seems that you can manage your expenses really well! Keep it up, " + name + "!"); } // thank you message System.out.println("\nThank you for using the Personal Expense Tracker, " + name + "!"); } public static void main(String[] args) { Scanner scanner = new Scanner(System.in); // call the title method showTitle(); // ask for the user's name System.out.print("\nEnter your name: "); String name = scanner.nextLine(); // ask for the budget System.out.print("Enter your budget: Php"); double budget = scanner.nextDouble(); // ask for each expense System.out.print("Enter Food expense: Php"); double food = scanner.nextDouble(); System.out.print("Enter Transportation expense: Php"); double transportation = scanner.nextDouble(); System.out.print("Enter Other expense: Php"); double other = scanner.nextDouble(); // extra expense category System.out.print("Enter Savings: Php"); double savings = scanner.nextDouble(); // call the compute method double total = computeTotal(food, transportation, other, savings); // call the display method showResult(name, food, transportation, other, savings, total, budget); scanner.close(); } }
Reflection
The most complex activity of the midterm — and my favorite. It combined everything: classes, arrays, methods, and control flow into one cohesive program. Perfect score, and the one I'm most proud of.
